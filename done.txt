Inicialização do projeto com o composer

Configuração no namespace PSR-4 no composer.json

{
    "autoload": {
        "psr-4": {
            "App\\": "src/"
        }
    },
    "name": "gidenilson/survey",
    "description": "Aplicativo de votação simples.",
    "authors": [
        {
            "name": "Gidenilson Santiago",
            "email": "gidenilson@gmail.com"
        }
    ],
    "require": {
        "bramus/router": "^1.6"
    }
}

Arquivo public/index.php

<?php

require __DIR__ . "../../vendor/autoload.php";

// Instanciamento do router
$router = new \Bramus\Router\Router();

// Definição das rotas
$router->get('/', 'App\Controllers\HomeController@index');

// Roda o router!
$router->run();
